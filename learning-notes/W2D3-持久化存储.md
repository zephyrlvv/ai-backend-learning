# W2D3：持久化存储（SQLite）

**目标**：把对话历史从内存保存到数据库，重启后数据不丢失

---

## 1. SQLite 是什么？

**一句话**：Python 内置的轻量级数据库，数据存在**单个文件**里，不需要安装服务器。

**类比**：
- 内存列表（现在）= 浏览器的 `sessionStorage`，关闭页面就清空
- SQLite = `localStorage`，永久保存，但功能更强大（支持 SQL 查询）

**为什么选 SQLite**：
- ✅ Python 内置，无需安装
- ✅ 单个文件，方便备份迁移
- ✅ 学习简单，W4 项目会用到更复杂的 PostgreSQL

## 2. Python 操作 SQLite 基础

### 连接数据库（文件不存在会自动创建）

```python
import sqlite3

# 连接数据库（chat.db 是文件名）
conn = sqlite3.connect('chat.db')

# 创建游标（执行 SQL 用的）
cursor = conn.cursor()
```

### 创建表

```python
# 创建对话历史表
cursor.execute('''
    CREATE TABLE IF NOT EXISTS messages (
        id INTEGER PRIMARY KEY AUTOINCREMENT,  -- 自增ID
        role TEXT NOT NULL,                     -- 'user' 或 'assistant'
        content TEXT NOT NULL,                  -- 消息内容
        session_id TEXT DEFAULT 'default',      -- 会话ID（支持多会话）
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP  -- 创建时间
    )
''')

# 提交更改
conn.commit()
```

**SQL 类型对比 TypeScript**：

| SQLite 类型 | TypeScript 类比 | 说明 |
|-------------|----------------|------|
| `INTEGER` | `number` | 整数 |
| `TEXT` | `string` | 字符串 |
| `TIMESTAMP` | `Date` | 时间戳 |
| `PRIMARY KEY` | 主键 | 唯一标识 |
| `AUTOINCREMENT` | 自增 | 自动+1 |

### 插入数据

```python
# 插入单条
cursor.execute(
    "INSERT INTO messages (role, content, session_id) VALUES (?, ?, ?)",
    ('user', '你好', 'default')
)
conn.commit()

# 批量插入
messages = [
    ('user', '你好', 'default'),
    ('assistant', '你好！有什么可以帮助你？', 'default')
]
cursor.executemany(
    "INSERT INTO messages (role, content, session_id) VALUES (?, ?)",
    messages
)
conn.commit()
```

**注意**：用 `?` 占位符防止 SQL 注入（类似前端防止 XSS）

### 查询数据

```python
# 查询所有消息
cursor.execute("SELECT * FROM messages ORDER BY created_at")
rows = cursor.fetchall()  # 返回列表 [(1, 'user', '你好', ...), ...]

# 查询最近10条
cursor.execute(
    "SELECT * FROM messages WHERE session_id = ? ORDER BY created_at DESC LIMIT ?",
    ('default', 10)
)
rows = cursor.fetchall()

# 转换为字典（更方便）
cursor.row_factory = sqlite3.Row  # 让查询结果可以用列名访问
rows = cursor.execute("SELECT * FROM messages").fetchall()
for row in rows:
    print(row['role'], row['content'])
```

### 删除数据

```python
# 清空表
cursor.execute("DELETE FROM messages")
conn.commit()

# 删除单条
cursor.execute("DELETE FROM messages WHERE id = ?", (1,))
conn.commit()
```

### 关闭连接

```python
conn.close()
```

## 3. 完整代码改造

### 新建 database.py（数据库操作模块）

在 `G:\kimiProject-pc` 下创建 `database.py`：

```python
import sqlite3
from typing import List, Dict
from contextlib import contextmanager

DATABASE_FILE = 'chat.db'

@contextmanager
def get_db():
    """上下文管理器，自动管理连接关闭"""
    conn = sqlite3.connect(DATABASE_FILE)
    conn.row_factory = sqlite3.Row  # 让结果可以用列名访问
    try:
        yield conn
    finally:
        conn.close()

def init_db():
    """初始化数据库，创建表"""
    with get_db() as conn:
        cursor = conn.cursor()
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS messages (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                role TEXT NOT NULL,
                content TEXT NOT NULL,
                session_id TEXT DEFAULT 'default',
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        ''')
        conn.commit()

def add_message(role: str, content: str, session_id: str = 'default'):
    """添加消息"""
    with get_db() as conn:
        cursor = conn.cursor()
        cursor.execute(
            "INSERT INTO messages (role, content, session_id) VALUES (?, ?, ?)",
            (role, content, session_id)
        )
        conn.commit()

def get_messages(session_id: str = 'default', limit: int = 20) -> List[Dict]:
    """获取最近的消息"""
    with get_db() as conn:
        cursor = conn.cursor()
        rows = cursor.execute(
            """SELECT role, content FROM messages 
               WHERE session_id = ? 
               ORDER BY created_at DESC LIMIT ?""",
            (session_id, limit)
        ).fetchall()
        
        # 返回列表 [{'role': 'user', 'content': 'xxx'}, ...]
        # 注意：要倒序回来（数据库是 DESC，AI 要 ASC）
        return [{'role': row['role'], 'content': row['content']} 
                for row in reversed(rows)]

def clear_messages(session_id: str = 'default'):
    """清空消息"""
    with get_db() as conn:
        cursor = conn.cursor()
        cursor.execute(
            "DELETE FROM messages WHERE session_id = ?",
            (session_id,)
        )
        conn.commit()

def get_stats() -> Dict:
    """获取统计信息"""
    with get_db() as conn:
        cursor = conn.cursor()
        total = cursor.execute(
            "SELECT COUNT(*) as count FROM messages"
        ).fetchone()['count']
        
        sessions = cursor.execute(
            "SELECT COUNT(DISTINCT session_id) as count FROM messages"
        ).fetchone()['count']
        
        return {'total_messages': total, 'total_sessions': sessions}
```
